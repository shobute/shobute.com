In today's world, having secure unique passwords1 is essential.

" We want Vim, not Vi.
set nocompatible

" Encrypt the file.
set cm=blowfish2

" Folding.
set foldexpr=getline(v:lnum)=~'^\\s*$'&&getline(v:lnum+1)=~'\\S'?'<1':1
set fdm=expr
set fdn=1
set fcl=all
set fdl=0
set foldlevelstart=0 foldenable

" Ignore case when searching.
set ignorecase

" A colour scheme.
set guioptions=
colorscheme desert

" Quit vim if idle for 5 minutes.
set updatetime=300000
autocmd CursorHold,CursorHoldI * q!

" No leakage.
set nobackup
set nowritebackup
set noundofile
set noswapfile
set viminfo=""
set noshelltemp
set history=0
set nomodeline
set secure

1 I prefer the term "passphrase" over "password", as "word" is often assumed to mean something short.
